Firestore Security Rules - Required Configuration

Go to Firebase Console -> Firestore Database -> Rules

Replace with these rules:

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Users collection
    match /users/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Conversations collection
    match /conversations/{conversationId} {
      allow read: if request.auth != null && 
                     request.auth.uid in resource.data.memberIds;
      allow create: if request.auth != null && 
                       request.auth.uid in request.resource.data.memberIds;
      allow update: if request.auth != null && 
                       request.auth.uid in resource.data.memberIds;
      allow delete: if false;
    }
    
    // Messages subcollection
    match /conversations/{conversationId}/messages/{messageId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null;
      // Allow update for:
      // 1. Message sender (for recall functionality)
      // 2. Any authenticated user (for adding/removing reactions)
      allow update: if request.auth != null && 
                       (request.auth.uid == resource.data.senderId ||
                        onlyReactionsChanged());
      allow delete: if false;
    }
    
    // Helper function to check if only reactions field changed
    function onlyReactionsChanged() {
      let affectedKeys = request.resource.data.diff(resource.data).affectedKeys();
      return affectedKeys.hasOnly(['reactions']) || 
             affectedKeys.hasOnly(['reactionCounts']) ||
             affectedKeys.hasOnly(['reactions', 'reactionCounts']);
    }
    
    // Friend requests collection
    match /friendRequests/{requestId} {
      // Allow reading own requests (received or sent)
      allow read: if request.auth != null && 
                     (request.auth.uid == resource.data.fromUserId || 
                      request.auth.uid == resource.data.toUserId);
      // Allow querying accepted requests (needed for getFriends)
      allow list: if request.auth != null;
      allow create: if request.auth != null && 
                       request.auth.uid == request.resource.data.fromUserId;
      allow update: if request.auth != null && 
                       request.auth.uid == resource.data.toUserId;
      allow delete: if false;
    }
  }
}

IMPORTANT: 
1. Copy all rules above
2. Go to Firebase Console -> Firestore -> Rules tab
3. Replace ALL existing rules with the rules above
4. Click "Publish"
5. Wait 1-2 minutes for rules to propagate
6. Test app again
